let company = { sales: [ {name: 'John', salary: 1000}, {name: 'Alice', salary: 600} ], development: { web: [ {name: 'Peter', salary: 2000}, {name: 'Alex', salary: 1800} ], internals: [ {name: 'Jack', salary: 1300} ] } }; function calculateTotalSalary(department) { let total = 0; if (Array.isArray(department)) { // підсумовуємо зарплати всіх співробітників for (let employee of department) { total += employee.salary; } } else { for (let subDepartment of Object.values(department)) { total += calculateTotalSalary(subDepartment); } } return total; } console.log(calculateTotalSalary(company)); // cумa всіх зарплат
